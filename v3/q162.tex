\begin{problem}
  Suppose $f(a)$ is the area of the region bounded by $x^2 + y^2 \leq 25$ and $-5 \leq x \leq a$.  What integer is closest to $\frac{f(4.01) - f(4)}{0.01}$?
  \begin{multipleChoice}
    \choice{3}
    \choice{4}
    \choice{5}
    \choice[correct]{6}
  \end{multipleChoice}
\end{problem}
