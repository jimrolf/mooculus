\begin{problem}
  Suppose $x = \frac{5 + \sqrt{0.1}}{200}$ and $y = \frac{21}{10}$.  What integer is closest to $x - y$?
  \begin{multipleChoice}
    \choice{$-26$}
    \choice{$-21$}
    \choice{$-16$}
    \choice[correct]{$-2$}
    \choice{$200$}
  \end{multipleChoice}
\end{problem}
