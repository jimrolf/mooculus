\begin{problem}
  The expression $\log_b(x) = y$ is equivalent to:
  \begin{multipleChoice}
    \choice{$b^x = y$}
    \choice[correct]{$b^y = x$}
    \choice{$x^b = y$}
    \choice{$x^y = b$}
    \choice{$y^b = x$}
    \choice{$y^x = b$}
  \end{multipleChoice}  
\end{problem}
