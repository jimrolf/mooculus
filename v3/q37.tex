\begin{problem}
	Suppose $x$ is a small negative number.  What can be said about $\frac{x}{|x|}$?
	\begin{multipleChoice}
		\choice[correct]{It is close to $-1$}
		\choice{It is close to $1$}
		\choice{It is close to $0$}
	\end{multipleChoice}
\end{problem}
