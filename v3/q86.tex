\begin{problem}
  Let $f(x) = x^{800}$.  How does $A = f(10^{10} + 1) - f(10^{10})$ compare to $B = f(10^{10}) - f(10^{10} - 1)$?
  \begin{multipleChoice}
    \choice{$A < B$}
    \choice{$A = B$}
    \choice[correct]{$A > B$}
  \end{multipleChoice}
\end{problem}
