\begin{problem}
  Suppose $p(x) = x^2 + a$ is a quadratic polynomial.  If $a > 0$, is it possible that there is a point $x$ so that $p'(x) = p(x) = 0$?
  \begin{multipleChoice}
    \choice{Yes}
    \choice[correct]{No}
  \end{multipleChoice}
\end{problem}
