\begin{problem}
  Suppose $f$ is a differentiable function and $f(0) = 0$ and
  $f(\sin 0.03) = 6$.  What is a best guess for $f(0.01)$?

    \choice{$4$}
    \choice{$6$}
  \end{multipleChoice}
